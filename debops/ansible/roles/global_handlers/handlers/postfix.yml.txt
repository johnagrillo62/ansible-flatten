~d0.name = Process Postfix Makefile
~d1.environment = 
~d2.LC_MESSAGES = 'C'
~d1.ansible.builtin.command = make
~d1.args = 
~d2.chdir = '/etc/postfix'
~d1.register = global_handlers__postfix_register_make
~d1.notify = [ 'Check postfix and reload' ]
~d1.changed_when = 'not global_handlers__postfix_register_make.stdout.startswith("make: Nothing to be done")'
~d0.name = Check postfix and restart
~d1.ansible.builtin.command = /usr/sbin/postfix -c /etc/postfix check
~d1.register = global_handlers__postfix_register_check_restart
~d1.changed_when = global_handlers__postfix_register_check_restart.changed | bool
~d1.notify = [ 'Restart postfix' ]
~d0.name = Check postfix and reload
~d1.ansible.builtin.command = /usr/sbin/postfix -c /etc/postfix check
~d1.register = global_handlers__postfix_register_check_reload
~d1.changed_when = global_handlers__postfix_register_check_reload.changed | bool
~d1.notify = [ 'Reload postfix' ]
~d0.name = Restart postfix
~d1.ansible.builtin.service = 
~d2.name = 'postfix'
~d2.state = 'restarted'
~d0.name = Reload postfix
~d1.ansible.builtin.service = 
~d2.name = 'postfix'
~d2.state = 'reloaded'
